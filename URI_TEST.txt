# Standard URI Examples (http, https, ftp)
# Full format with all components
http://user@example.com:8080/path/to/resource?query=value#fragment
https://admin@server.domain.com:443/docs/file.txt?version=2#section1
ftp://ftp-user@ftp.storage.com:21/downloads/data.zip

# Without userinfo
http://example.com/path
https://api.service.com:8443/v1/endpoint

# Without port (defaults to 80)
http://user@example.com/path
https://domain.com/resource

# With IP address as host
http://192.168.1.1/admin
https://10.0.0.1:8080/settings

# Minimal valid URIs
http://example.com
https://domain.org
ftp://files.com

# Special Scheme: mailto
# Full format
mailto:user@domain.com
mailto:admin@server.company.com

# Without host
mailto:username

# Special Scheme: news
news:comp.programming
news:server.news.com

# Special Scheme: tel/fax
tel:+1234567890
tel:00390123456789
fax:+1122334455

# Special Scheme: zos
# Basic format with valid id44
zos://mainframe.ibm.com/ABC.DEF.GHI

# With id44 and id8
zos://system1.company.com/PROGRAM.SOURCE.CODE(MAIN)

# With port and other components
zos://mainframe:2323/MY.DATASET(DATA)?version=1#section2

# Complex examples with multiple components
zos://user@mainframe.company.com:1047/SYSTEM.DATA.SET(MEMBER)?mode=binary#top

# Examples with query and fragment
http://service.com/api?key=value&mode=test#section
https://docs.example.com/guide?lang=en&version=2#introduction

# Minimal paths
http://domain.com/file
https://api.com/endpoint

# Multi-level paths
http://server.com/level1/level2/level3
https://platform.org/api/v1/users/profile

# Invalid Scheme Format
@http://example.com            # scheme non può iniziare con caratteri speciali
123://example.com             # scheme deve iniziare con una lettera
http+s://example.com          # caratteri speciali non permessi nello scheme

# Invalid Authority Format
http:///example.com           # triplo slash non permesso
http:/example.com             # singolo slash non valido (deve essere // o nessuno)
http:/@example.com            # userinfo senza host non permesso
http://user:pass@example.com  # password non prevista nella grammatica

# Invalid Host Format
http://example..com           # doppio punto non permesso
http://.example.com           # host non può iniziare con punto
http://example-.com          # caratteri speciali non permessi nel hostname
http://exam&ple.com          # caratteri speciali non permessi nel hostname

# Invalid IP Address Format
http://256.1.2.3             # numeri > 255 non permessi
http://1.2.3                 # IP incompleto
http://1.2.3.4.5             # troppi numeri nell'IP
http://1.2.3.abc             # caratteri non numerici nell'IP

# Invalid Port Format
http://example.com:65536     # porta fuori range
http://example.com:abc       # porta deve essere numerica
http://example.com:-80       # porta non può essere negativa
http://example.com:8080:90   # multiple porte non permesse

# Invalid Path Format
http://example.com/path@     # caratteri speciali non permessi nel path
http://example.com/pa#th     # fragment deve essere alla fine
http://example.com//path     # doppio slash non permesso nel path

# Invalid Query Format
http://example.com?          # query vuota non permessa
http://example.com/path??query  # doppio ? non permesso
http://example.com?#         # query vuota con fragment non permessa

# Invalid Fragment Format
http://example.com#         # fragment vuoto non permesso
http://example.com##        # doppio # non permesso

# Invalid mailto Schema
mailto:@domain.com          # userinfo vuoto non permesso
mailto:user@               # host vuoto non permesso
mailto:user@@domain.com    # doppia @ non permessa

# Invalid news Schema
news://groups.google.com    # // non permesso per news
news:@group                # @ non permesso per news
news:group@domain          # @ non permesso per news

# Invalid tel/fax Schema
tel://1234567              # // non permesso per tel
tel:@1234567              # @ non permesso per tel
fax://1234567             # // non permesso per fax

# Invalid zos Schema
# Violazioni id44
zos://host/               # path vuoto non permesso
zos://host/.DATA         # non può iniziare con punto
zos://host/1DATA         # deve iniziare con lettera
zos://host/ABCDEF.       # non può terminare con punto
zos://host/TOOLONGNAME.EXCEEDING.FORTY.FOUR.CHARACTERS.IN.TOTAL  # > 44 caratteri

# Violazioni id8
zos://host/DATA(1MEM)    # id8 deve iniziare con lettera
zos://host/DATA(TOOLONGMEMBER)  # id8 > 8 caratteri
zos://host/DATA()        # id8 vuoto non permesso
zos://host/DATA(MEM@)    # caratteri speciali non permessi in id8
